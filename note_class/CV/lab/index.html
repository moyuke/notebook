cap = cv2.VideoCapture('sfm.mp4')
while(cap.isOpened()):
    ret, frame = cap.read()
    if ret == True:
        cv2.imshow('Frame',frame)
        if cv2.waitKey(25) & 0xFF == 27:
            # 按esc退出
            break
    else:
        break
cap.release()
# 播放视频
